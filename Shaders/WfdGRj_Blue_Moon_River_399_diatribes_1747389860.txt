#define A(x,r) abs(dot(sin(x*p*a), r +p-p )) / a

void mainImage(out vec4 o, vec2 u) 
{
    float i,r,d,s,a,t=iTime*5e1;
    vec3  p = iResolution; 
    
    u = (u-p.xy/2.)/p.y;
    
    for (o*=i;i++<1e2; 
         d += s = .02 + abs(min(s,r))*.2,
         o += vec4(1,2,4,0) / s)
         
        for(p = vec3(u * d, d + t),
            a = .01,
            r = 15. + p.y,
            s = r + 7. - sin(p.z*.01) * 32. -
            abs(p.x += sin(p.z*.03)*8.)*.72;

            a < .1;

            r += A(.06 * p.z + .04 * t + 12., .015),
            s += A(1.,.5),

            a *= 1.4);

    o = tanh(o / 2e3 / dot(u-=.45,u));
}
