#define SCALE 6.
#define T iTime*40.
#define N normalize
#define P(z) (vec3((tanh(cos((z) * .01) *1.) * 42.), \
                   (cos((z) * .005) * 64.), (z)))
#define rot(a) mat2(cos(a), -sin(a), sin(a), cos(a))


float tunnels(vec3 p) {
    vec2 t = P(p.z).xy;
    float s = 16. -
               min(length(p.y - t.x),
               min(length(p.xy - t.x),
               min(length(p.xy - t.xy),
                   length(p.xy - t.y))));
    return s;
}

// I based this on @chronos' "Lode Style Voxel DDA 3D"
// https://www.shadertoy.com/view/X32SD3
vec3 tex(vec3 ro, vec3 rd, vec3 p, vec3 n, float side) {
    vec3 rgb;                         
    
    // Solve ray plane intersection equation: dot(n, ro + t * rd - p = 0.
    // for t 
    float t = dot(n, p - ro) / dot(n, N(rd));
    vec3 hit = ro + N(rd) * t;
 
    if(side == 0.)
        rgb = texture(iChannel0, hit.yz*.05).rgb,
        rgb.r += .15;
    else if(side == 1.)
        rgb = texture(iChannel1, hit.xz*.05).rgb*6.,
        rgb.r += .15;
    else
        rgb = texture(iChannel2, hit.xy*.05).rgb,
        rgb.r += .01;
    return rgb;
}

void mainImage(out vec4 o, vec2 u) {

    vec3  id,
          r = iResolution;
    u = (u-r.xy/2.)/r.y;
          // rayPos
    vec3  p = P(T)/SCALE, ro = p,
          Z = N( P(T+20.) - P(T)),
          X = N(vec3(Z.z,0,-Z)),
          
          // rayDir
          rd = vec3(rot(tanh(sin(p.z*.02)*4.)*3.)*u, 1) 
             * mat3(-X, cross(X, Z), Z),
          fp = floor(p);
              
    // rayStep
	vec3 rs = sign(rd),
         // deltaDist
	     dd = 1. / abs(rd),
         // sideDist
		 sd = (rs * (fp - p) + (rs * 0.5) + 0.5) * dd; 

    o-=o;
    float i,s,d;
    float side = 0.;
	for (i=0.; i++ < 160.;) {

        if(sd.x <= sd.y && sd.x <= sd.z){
            sd.x += dd.x; // Add delta dist
            fp.x += rs.x; // step right/left
            side = 0.;    // yz wall
        } else if(sd.y <= sd.z && sd.y <= sd.x) {
            sd.y += dd.y; // Add delta dist
            fp.y += rs.y; // step up/down
            side = 1.;    // xz wall
        }  else {
            sd.z += dd.z; // Add delta dist
            fp.z += rs.z; // step front/back
            side = 2.;    // xy wall
        }
        s = tunnels(fp*SCALE);
        id = fp/s;
        d +=s;
		if (s < 0.) break;
	}
   
    p = fp  + .5 - rs * .5;
    vec3 n = vec3(side==0., side==1., side==2.);
    o.rgb = vec3(.8,.6,.1)*tex(ro, rd, p, n, side);
    o.rgb *= max(dot(n, N(rd)), .075)*1.5;
    o.rgb = pow(o.rgb*exp(-(i*.03)), vec3(.45));
}
