// ===== PASS 0: Image =====

void mainImage(out vec4 o, vec2 u) 
{
    float i,d;
    vec3 p = iResolution;
    u = (u-p.xy/2.)/p.y;
    for(o*=i; i++<60.; o += 1.+cos(p.zxxy))
        p = vec3(u * d, d),
        p.xy *= mat2(cos(iTime+.3*p.z+vec4(0,33,11,0))),
        d += .005+.3*(abs(fract(p.x)-.5)+1.+cos(p.y));
    o = tanh(o*o*o/d/2e4);
}
