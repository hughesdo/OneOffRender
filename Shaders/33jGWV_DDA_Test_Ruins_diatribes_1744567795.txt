#define SCALE 12.
#define GROUND 164.
#define T iTime*152.
#define N normalize
#define P(z) (vec3(tanh(cos((z)*.002) *  1.6) * 384., \
                    tanh(cos((z) * .0025) *2.1) * GROUND, (z)))

#define rot(a) mat2(cos(a), -sin(a), sin(a), cos(a))

float side = 0.;

float tunnels(vec3 p) {
    vec2 t = P(p.z).xy;
    vec2 sq = abs(p.xy - t);
    float s, s1, s2;

    s1 = min(length(p.x - t.y),
         min(length(p.xy - t.xy) - 15.,
             length(p.xy - t.y ) + 5.));

    s = 123.- min(length(p.y - t.x), s1);
    s = min(s, max(72. - sq.x, 32. - sq.y));
    return s;
}

// I based this on @chronos' "Lode Style Voxel DDA 3D"
// https://www.shadertoy.com/view/X32SD3
vec3 tex(vec3 ro, vec3 rd, vec3 p, vec3 n) {
    vec3 rgb;                         
    
    // Solve ray plane intersection equation: dot(n, ro + t * rd - p = 0.
    // for t 
    float t = dot(n, p - ro) / dot(n, N(rd));
    vec3 hit = ro + N(rd) * t;
 
    if(side == 0.)
        rgb = texture(iChannel0, hit.yz*.125).rgb;
    else if(side == 1.)
        rgb = texture(iChannel1, hit.xz*.0675).rgb*3.,
        rgb.g += (sin(p.z*.03)*.45+.65)*1.,
        rgb.r += (sin(p.z*.045)*.35+.25)*1.1;
    else
        rgb = texture(iChannel2, hit.xy*.0275).rgb*4.;
  
    // noise texture
    for (t = .2; t < 1.2;
        rgb.rgb -= abs(dot(sin(rgb * t * 32.), vec3(.05))) / t,
        t *= 1.4142);
   
    rgb *= rgb * .35;
    return rgb;
}

float ground(vec3 p) {
    float s = GROUND - p.y;
    return s;
}

vec4 marchVoxels(vec3 fp, vec3 sd, vec3 rs, vec3 dd) {

    vec3 q;
    float i,os=0.,s,d=0.;
	for (i=0.; i++ < 170.;) {

        if(sd.x <= sd.y && sd.x <= sd.z){
            sd.x += dd.x; // Add delta dist
            fp.x += rs.x; // step right/left
            side = 0.;    // yz wall
        } else if(sd.y <= sd.z && sd.y <= sd.x) {
            sd.y += dd.y; // Add delta dist
            fp.y += rs.y; // step up/down
            side = 1.;    // xz wall
        }  else {
            sd.z += dd.z; // Add delta dist
            fp.z += rs.z; // step front/back
            side = 2.;    // xy wall
        }
        
        // tunnels to carve out geometry
        s = tunnels(fp*SCALE);
        
        // ground is just a plane
        s = min(s, ground(fp*SCALE));
        
        // todo: add water :)
        d += s;
		if (s < 0.) break;

	}
    
    return vec4(fp, i);
}

void mainImage(out vec4 o, vec2 u) {

    vec3  id,
          green = vec3(1,2,1)*.5,
          r = iResolution;
    u = (u-r.xy/2.)/r.y;
          // rayPos
    vec3  p = P(T)/SCALE, ro = p,
          Z = N( P(T+16.) - P(T)),
          X = N(vec3(Z.z,0,-Z)),
          
          // rayDir
          rd = vec3(rot(sin(T*.001)*.3)*u, 1) 
             * mat3(-X, cross(X, Z), Z);
              
    // rayStep
	vec3 rs = sign(rd),
         // deltaDist
	     dd = 1. / abs(rd),
         // sideDist
		 sd = (rs * (floor(p) - p) + (rs * 0.5) + 0.5) * dd; 

    o-=o;

    vec4 vox = marchVoxels(floor(p), sd, rs, dd);
    //vec4 orb = marchOrbs(ro, rd);

    vec3 n = vec3(side==0., side==1., side==2.);
    p = vox.xyz  + .5 - rs * .5,
    o.rgb = green*tex(ro, rd, p, n),
    o.rgb = pow(o.rgb*exp(-vox.w/64.)-dot(u,u)*.25, vec3(.45));
}
